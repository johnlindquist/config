#!/usr/bin/env bash
# Pin/unpin the currently‑focused window on the left edge.

PIN_W=400                             # sidebar width in px
win_id=$(yabai -m query --windows --window | jq '.id')
is_float=$(yabai -m query --windows --window | jq -r '.floating')

if [[ "$is_float" == "off" ]]; then
  # --- pin it ---
  yabai -m window $win_id --toggle float                    # unmanaged now
  yabai -m window $win_id --move   abs:0:0                  # hug left edge
  yabai -m window $win_id --resize abs:$PIN_W:0             # full height, fixed width
  yabai -m space  --padding abs:0:0:$PIN_W:0                # shrink BSP grid
else
  # --- un‑pin it ---
  yabai -m window $win_id --toggle float
  yabai -m space  --padding abs:0:0:0:0                     # restore full width
fi


---

Execute the following Git Commit Steps

1. Check current branch:
   git branch --show-current | cat
   BRANCH CHECK!
      - Ensure you're on the intended branch
      - If wrong branch: git checkout <correct-branch> and restart

2. Stage all changes:
   git add .

3. Review staged changes:
   git status | cat
   SAFETY CHECKS!
      - Look for .env files, secrets, or other sensitive files that should not be committed
      - Check for large files: git diff --staged --stat | cat
      - If found: git reset and STOP

4. Review actual changes:
   git diff --staged | cat
   CODE REVIEW!
      - Verify all changes are intentional
      - Look for debug code, console.logs
      - Check for hardcoded values
      
5. Choose commit type:
   - For fixes: git commit -m "fix($scope): $message"
   - For features: git commit -m "feat($scope): $message"
   - Other types: build, chore, ci, docs, perf, refactor, revert, style, test
   
   Commit Format Rules:
   - Type must be lowercase
   - Scope is optional, must be in parentheses
   - Message must be imperative: "add" not "added"
   - No period at end of message
   
6. Push changes:
   git push

Examples:
fix: git add . && git commit -m "fix(auth): handle expired tokens" && git push
feat: git add . && git commit -m "feat(ui): add dark mode toggle" && git push
perf: git add . && git commit -m "perf(api): optimize database queries" && git push
